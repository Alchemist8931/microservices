SQL для таблицы пользователей (Supabase / Postgres)

Содержит общий ключ группы, роль, уникальный идентификатор пользователя, логин, пароль, дату последнего входа и баланс администратора.

CREATE EXTENSION IF NOT EXISTS \"pgcrypto\"; -- для gen_random_uuid()\n\nCREATE TABLE IF NOT EXISTS app_users (\n group_key uuid NOT NULL DEFAULT gen_random_uuid(), -- общий ключ группы пользователей (генерируется у администратора)\n role text NOT NULL CHECK (role IN ('admin', 'user')),\n user_id uuid PRIMARY KEY DEFAULT gen_random_uuid(), -- уникальный ID пользователя\n username text NOT NULL UNIQUE, -- логин\n password_hash text NOT NULL, -- храните хэш пароля\n last_login timestamptz,\n admin_balance numeric(14,2), -- только для admin\n CONSTRAINT admin_balance_required CHECK (\n (role = 'admin' AND admin_balance IS NOT NULL) OR\n (role = 'user' AND admin_balance IS NULL)\n )\n);\n\n-- Пользователи одного администратора делят общий group_key (значение админа передавайте при создании user)\n-- При необходимости ограничьте логины внутри группы:\n-- CREATE UNIQUE INDEX IF NOT EXISTS app_users_group_username_key ON app_users (group_key, username);\n
